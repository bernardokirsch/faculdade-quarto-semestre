1) Crie as tabelas de acordo com o modelo lógico apresentado.

CREATE TABLE EVENTO (
	IDEVENTO		SERIAL,
	NOME 			VARCHAR (100),
	EDICAO			VARCHAR (5),
	LUGAR			VARCHAR (100),
	PRIMARY KEY (IDEVENTO)
);

CREATE TABLE PARTICIPANTE (
	IDPART			SERIAL,
	NOME 			VARCHAR (100),
	CPF				VARCHAR (14),
	EMAIL			VARCHAR (75),
	PRIMARY KEY (IDPART)
);

CREATE TABLE PART_EVENTO (
	IDPART			INT,
	IDEVENTO 		INT,
	PRIMARY KEY (IDPART, IDEVENTO),
	FOREIGN KEY (IDPART) REFERENCES PARTICIPANTE (IDPART),
	FOREIGN KEY (IDEVENTO) REFERENCES EVENTO (IDEVENTO)
);

CREATE TABLE PALESTRA (
	IDPALESTRA		SERIAL,
	TITULO			VARCHAR (60),
	IDEVENTO		INT,
	PRIMARY KEY (IDPALESTRA),
	FOREIGN KEY (IDEVENTO) REFERENCES EVENTO (IDEVENTO)
);
